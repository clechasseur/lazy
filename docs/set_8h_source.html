27 /**
28  * @class coveo::lazy::set
29  * @extends coveo::lazy::detail::lazy_sorted_container
30  * @brief Set container that performs lazy sorting.
31  * @headerfile set.h <coveo/lazy/set.h>
32  *
33  * <tt>std::set</tt>-like container class that stores its elements in an internal
34  * container (by default, an <tt>std::vector</tt>) and sorts them only when needed.
35  * Sorting can also be triggered on-demand.
36  *
37  * See <tt>coveo::lazy::detail::lazy_sorted_container</tt> for details
38  * on the actual implementation.
39  *
40  * This class has the following differences compared to <tt>std::set</tt>:
41  *
42  * - Because it uses a sequence container internally, elements in the set might
43  * be moved/copied when an insertion occurs. Thus, this class cannot be used
44  * to store move-only types. Furthermore, inserting an element invalidates
45  * all iterators and references.
46  * - It needs a predicate to determine if two set elements are equal. If an
47  * <tt>operator==</tt> exists to compare instances of type @c K, it will be
48  * used; otherwise, it will use an implementation that uses @c key_compare
49  * to determine equality. This is less efficient; if a custom equality
50  * predicate exists for type @c K, it should be used for the @c _Eq template
51  * parameter.
52  * - Modifier methods do not return iterators. This is done to better
53  * support lazy sorting.
54  *
55  * @tparam K Type of elements stored in the set.
56  * @tparam _Cmp Predicate used to sort the elements.
57  * Defaults to <tt>std::less<K></tt>.
58  * @tparam _Impl Type of sequence container used by the set.
59  * Defaults to <tt>std::vector</tt>.
60  * @tparam _Eq Predicate used to determine if elements are equal.
61  * Defaults to using <tt>operator==</tt> if available,
62  * otherwise to an implementation that uses @c _Cmp (see above).
63  * @tparam _Alloc Allocator used for the set elements.
64  * Defaults to <tt>std::allocator</tt>.
65  */
66 template<class K,
67  class _Cmp = std::less<K>,
68  template<class _ImplT, class _ImplAlloc> class _Impl = std::vector,
69  class _Eq = detail::equal_to_using_less_if_needed<K, _Cmp>,
70  class _Alloc = std::allocator<K>>
71  using set = detail::lazy_sorted_container<K,
72  void,
73  K,
74  K,
75  detail::identity<K>,
76  _Cmp,
77  _Eq,
78  _Alloc,
79  _Impl,
80  false>;
